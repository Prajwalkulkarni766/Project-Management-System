delimiter $
create procedure sp_project(
        in p_userid int,
        in p_projectid int,
	in p_projectname varchar(20),
	in p_projectdescription varchar(1000),
	in p_mode tinyint,
	inout p_outparameter int
)
begin

-- mode 1 to insert the record of project
	if (p_mode = 1) then
		insert into tbl_project (project_name, project_description, user_id) values (p_projectname, p_projectdescription, p_userid);
		SET @p_outparameter = LAST_INSERT_ID();
	end if;

-- mode 2 to modify the project
	if (p_mode = 2) then
		update tbl_project set project_name = p_projectname , project_description = p_projectdescription where project_id = p_projectid;
		SET @p_outparameter = p_projectid;
	end if;

-- mode 3 to delete the project
	if (p_mode = 3) then
		delete from tbl_project where project_id = p_projectid;
		SET @p_outparameter = p_projectid;
	end if;

end $
delimiter ;
